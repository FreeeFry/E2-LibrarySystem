@name Simple keyPress example

if(first())
{
    #include "libraries/input/input_core"
    
    # Add the owner of the chip to the list of players to monitor input for.
    inputRegisterSource("owner", owner())
    
    # This will be called whenever a key press is detected.
    function myKeyPressedFunction(Player:entity, Key:string, State:number)
    {
        if(State) # Player pressed the key.
        {
            print(Player, "pressed", Key, "last state:", State)
        }
    }
    # Register the above function as a callback to the event so we can respond to key presses.
    registerCallback("event_playerInput", "myKeyPressedFunction")
}
