@name Small custom events example chip

if(first())
{
    #include "libraries/event_system/event_core"
    
    function event_PlayerPressUse(Callbacks:array) # This event detects whenever a player presses the use key and passes that to any registered callbacks.
    {
        local Players = players()
        for(I=1, Players:count())
        {
            local Player = Players[I, entity]
            if(Player:keyUse())
            {
                for(II=1, Callbacks:count())
                {
                    local StrCallback = Callbacks[II, string]
                    StrCallback(Players[I, entity]) # Passes player to callback
                }
            }
        }
    }
    registerEvent("event_PlayerPressUse", 25) # 
    
    function playerPressed(Player:entity)
    {
        print(Player, "is pressing the use key!")
    }
    registerCallback("event_PlayerPressUse", "playerPressed")
}
